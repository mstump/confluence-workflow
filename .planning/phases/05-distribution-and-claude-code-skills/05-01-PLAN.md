  <name>Task 1: License decision — GPL-3.0 vs MIT/Apache-2.0</name>
  <decision>Choose the license for the crates.io package</decision>
  <context>
The existing LICENSE file contains GPL-3.0. The Rust ecosystem convention is dual MIT/Apache-2.0.
GPL-3.0 is copyleft (derivative works must also be GPL-3.0). MIT/Apache-2.0 is permissive.
crates.io accepts either. The choice affects downstream users.

RESEARCH.md recommends `license = "MIT OR Apache-2.0"` for Rust ecosystem compatibility.
  </context>
  <options>
    <option id="option-a">
      <name>MIT OR Apache-2.0 (Rust ecosystem convention)</name>
      <pros>Permissive; standard in Rust ecosystem; maximum adoption; compatible with all downstream uses</pros>
      <cons>Replaces existing GPL-3 LICENSE file; less copyleft protection</cons>
    </option>
    <option id="option-b">
      <name>GPL-3.0 (keep existing)</name>
      <pros>No LICENSE file change needed; copyleft protection</pros>
      <cons>Non-standard for Rust crates; limits downstream proprietary use</cons>
    </option>
    <option id="option-c">
      <name>MIT only (simpler permissive)</name>
      <pros>Single license, single file; very permissive; widely understood</pros>
      <cons>No patent grant (Apache-2.0 adds that); not the dual-license Rust convention</cons>
    </option>
  </options>
  <resume-signal>Select: option-a, option-b, or option-c</resume-signal>
